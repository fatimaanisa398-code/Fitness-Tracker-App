this.prevBtn.addEventListener('click', () => this.prevCard()); this.nextBtn.addEventListener('click', () => this.nextCard()); this.addBtn.addEventListener('click', () => this.openModal('add')); this.editBtn.addEventListener('click', () => this.openModal('edit')); this.deleteBtn.addEventListener('click', () => this.deleteCard()); this.cardForm.addEventListener('submit', (e) => this.saveCard(e)); this.closeBtn.addEventListener('click', () => this.closeModal()); if (e.key === 'ArrowRight') this.nextCard(); if (e.key === 'Enter' && this.modal.classList.contains('active')) { this.cardForm.dispatchEvent(new Event('submit')); } if (e.key === 'Escape') this.closeModal(); }); } updateDisplay() { this.updateCounter(); this.updateButtons(); if (this.cards.length === 0) { this.questionEl.textContent = "Click 'Add Card' to create your first flashcard!"; this.cardFront.style.display = 'flex'; this.cardBack.style.display = 'none'; this.showAnswerBtn.style.display = 'none'; } else { this.showCard(this.currentIndex); } this.saveToStorage(); } showCard(index) { const card = this.cards[index]; this.questionEl.textContent = card.question; this.answerEl.textContent = card.answer; this.isShowingAnswer = false; this.cardFront.style.display = 'flex'; this.cardBack.style.display = 'none'; this.showAnswerBtn.style.display = 'block'; this.showAnswerBtn.textContent = 'Show Answer'; this.showAnswerBtn.innerHTML = ' Show Answer'; } toggleAnswer() { if (this.cards.length === 0) return; this.isShowingAnswer = !this.isShowingAnswer; if (this.isShowingAnswer) { this.cardFront.style.display = 'none'; this.cardBack.style.display = 'flex'; this.showAnswerBtn.innerHTML = ' Hide Answer'; } else { this.cardFront.style.display = 'flex'; this.cardBack.style.display = 'none'; this.showAnswerBtn.innerHTML = ' Show Answer'; } } nextCard() { if (this.cards.length === 0) return; this.currentIndex = (this.currentIndex + 1) % this.cards.length; this.showCard(this.currentIndex); } prevCard() { if (this.cards.length === 0) return; this.currentIndex = (this.currentIndex - 1 + this.cards.length) % this.cards.length; this.showCard(this.currentIndex); } updateCounter() { this.counterEl.textContent = `Card ${this.currentIndex + 1} of ${this.cards.length}`; } updateButtons() { this.prevBtn.disabled = this.cards.length === 0; this.nextBtn.disabled = this.cards.length === 0; this.editBtn.disabled = this.cards.length === 0; this.deleteBtn.disabled = this.cards.length === 0; } openModal(mode, index = null) { this.modal.classList.add('active'); this.mode = mode; this.editIndex = index; if (mode === 'add') { this.modalTitle.textContent = 'Add New Flashcard'; this.questionInput.value = ''; this.answerInput.value = ''; } else { this.modalTitle.textContent = 'Edit Flashcard'; const card = this.cards[index]; this.questionInput.value = card.question; this.answerInput.value = card.answer; } this.questionInput.focus(); } closeModal() { this.modal.classList.remove('active'); } saveCard(e) { e.preventDefault(); const question = this.questionInput.value.trim(); const answer = this.answerInput.value.trim(); if (!question || !answer) { alert('Please fill in both question and answer!'); return; } if (this.mode === 'add') { this.cards.push({ question, answer }); this.currentIndex = this.cards.length - 1; } else { this.cards[this.editIndex] = { question, answer }; } this.updateDisplay();
